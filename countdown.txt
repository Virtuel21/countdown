    <script>
        // Fonction de d√©bogage
        function debugLog(message, type = 'info') {
            console.log(message);
            const debugDiv = document.getElementById('debug-output');
            const statusClass = type === 'success' ? 'success' : type === 'error' ? 'error' : '';
            debugDiv.innerHTML += `<p><span class="status ${statusClass}">${type.toUpperCase()}</span> ${message}</p>`;
        }
        
        // Clear initial loading message
        document.getElementById('debug-output').innerHTML = '';
        
        // Fonction pour initialiser le countdown
        function initializeCountdown() {
            debugLog('üöÄ D√©but de l\'initialisation du countdown', 'info');
            
            // Fonction pour mettre √† jour un countdown
            function updateCountdown(countdownElement) {
                const endDateStr = countdownElement.getAttribute('data-countdown-end');
                if (!endDateStr) {
                    debugLog('‚ùå Attribut data-countdown-end manquant', 'error');
                    return;
                }
                
                debugLog(`üìÖ Date de fin trouv√©e: ${endDateStr}`, 'info');
                
                const endDate = new Date(endDateStr).getTime();
                const now = new Date().getTime();
                const distance = endDate - now;
                
                // S√©lectionner les √©l√©ments
                const daysEl = countdownElement.querySelector('[data-days]');
                const hoursEl = countdownElement.querySelector('[data-hours]');
                const minutesEl = countdownElement.querySelector('[data-minutes]');
                const secondsEl = countdownElement.querySelector('[data-seconds]');
                
                if (!daysEl || !hoursEl || !minutesEl || !secondsEl) {
                    debugLog('‚ùå √âl√©ments du countdown non trouv√©s', 'error');
                    return;
                }
                
                if (distance < 0) {
                    // Countdown termin√©
                    daysEl.textContent = '00';
                    hoursEl.textContent = '00';
                    minutesEl.textContent = '00';
                    secondsEl.textContent = '00';
                    debugLog('‚è∞ Countdown termin√©', 'error');
                    return false;
                }
                
                // Calculer les unit√©s de temps
                const days = Math.floor(distance / (1000 * 60 * 60 * 24));
                const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
                const seconds = Math.floor((distance % (1000 * 60)) / 1000);
                
                // Mettre √† jour l'affichage
                daysEl.textContent = String(days).padStart(2, '0');
                hoursEl.textContent = String(hours).padStart(2, '0');
                minutesEl.textContent = String(minutes).padStart(2, '0');
                secondsEl.textContent = String(seconds).padStart(2, '0');
                
                return true;
            }
            
            // Trouver tous les countdowns
            const countdowns = document.querySelectorAll('.snrs-countdown[data-countdown-end]');
            debugLog(`üîç Nombre de countdowns trouv√©s: ${countdowns.length}`, countdowns.length > 0 ? 'success' : 'error');
            
            if (countdowns.length === 0) {
                debugLog('‚ùå Aucun countdown trouv√© avec le s√©lecteur .snrs-countdown[data-countdown-end]', 'error');
                return;
            }
            
            // Initialiser chaque countdown
            countdowns.forEach((countdown, index) => {
                debugLog(`‚úÖ Initialisation du countdown ${index + 1}`, 'success');
                
                // Premi√®re mise √† jour imm√©diate
                const isActive = updateCountdown(countdown);
                
                if (isActive) {
                    debugLog(`‚è±Ô∏è Timer actif - Mise √† jour chaque seconde`, 'success');
                    // Mettre √† jour chaque seconde
                    setInterval(() => {
                        updateCountdown(countdown);
                    }, 1000);
                }
            });
            
            // Afficher l'heure actuelle et la date de fin
            const now = new Date();
            const endDate = new Date('2025-12-06 23:59:59');
            debugLog(`üïê Heure actuelle: ${now.toLocaleString('fr-FR')}`, 'info');
            debugLog(`üéØ Date de fin: ${endDate.toLocaleString('fr-FR')}`, 'info');
            debugLog(`‚è≥ Temps restant: ${Math.floor((endDate - now) / (1000 * 60 * 60 * 24))} jours`, 'success');
        }
        
        // Lancer l'initialisation imm√©diatement
        debugLog('üìÑ DOM State: ' + document.readyState, 'info');
        
        if (document.readyState === 'loading') {
            debugLog('‚è≥ Attente du chargement du DOM...', 'info');
            document.addEventListener('DOMContentLoaded', initializeCountdown);
        } else {
            debugLog('‚úÖ DOM d√©j√† charg√©, initialisation imm√©diate', 'success');
            initializeCountdown();
        }
        
        // Test suppl√©mentaire apr√®s un court d√©lai
        setTimeout(function() {
            const countdownNumbers = document.querySelectorAll('.snrs-countdown-number');
            let hasValues = false;
            countdownNumbers.forEach(el => {
                if (el.textContent !== '00') hasValues = true;
            });
            
            if (hasValues) {
                debugLog('üéâ SUCC√àS: Le countdown fonctionne correctement!', 'success');
            } else {
                debugLog('‚ö†Ô∏è Les valeurs du countdown sont toujours √† 00', 'error');
            }
        }, 2000);
